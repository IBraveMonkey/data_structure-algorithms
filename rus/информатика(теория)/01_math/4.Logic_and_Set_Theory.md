## 4. ‚öñÔ∏è –ó–∞–∫–æ–Ω—ã –¥–µ –ú–æ—Ä–≥–∞–Ω–∞

## 5. üíª –†–µ–∞–ª–∏–∑–∞—Ü–∏—è –Ω–∞ Go

### Set (–ú–Ω–æ–∂–µ—Å—Ç–≤–æ) —á–µ—Ä–µ–∑ map

–í Go –Ω–µ—Ç –≤—Å—Ç—Ä–æ–µ–Ω–Ω–æ–≥–æ —Ç–∏–ø–∞ Set, –Ω–æ –º–æ–∂–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å `map[T]bool`:

```go
type Set map[int]bool

func NewSet(values ...int) Set {
    s := make(Set)
    for _, v := range values {
        s[v] = true
    }
    return s
}

func (s Set) Add(value int) {
    s[value] = true
}

func (s Set) Remove(value int) {
    delete(s, value)
}

func (s Set) Contains(value int) bool {
    return s[value]
}

func (s Set) Size() int {
    return len(s)
}

// –û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ (Union)
func (s Set) Union(other Set) Set {
    result := NewSet()
    for k := range s {
        result.Add(k)
    }
    for k := range other {
        result.Add(k)
    }
    return result
}

// –ü–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ (Intersection)
func (s Set) Intersection(other Set) Set {
    result := NewSet()
    for k := range s {
        if other.Contains(k) {
            result.Add(k)
        }
    }
    return result
}

// –†–∞–∑–Ω–æ—Å—Ç—å (Difference)
func (s Set) Difference(other Set) Set {
    result := NewSet()
    for k := range s {
        if !other.Contains(k) {
            result.Add(k)
        }
    }
    return result
}

func main() {
    a := NewSet(1, 2, 3, 4)
    b := NewSet(3, 4, 5, 6)
    
    fmt.Println("A:", a)                    // {1:true, 2:true, 3:true, 4:true}
    fmt.Println("B:", b)                    // {3:true, 4:true, 5:true, 6:true}
    
    union := a.Union(b)
    fmt.Println("A ‚à™ B:", union)            // {1, 2, 3, 4, 5, 6}
    
    intersection := a.Intersection(b)
    fmt.Println("A ‚à© B:", intersection)     // {3, 4}
    
    difference := a.Difference(b)
    fmt.Println("A \\ B:", difference)      // {1, 2}
}
```
