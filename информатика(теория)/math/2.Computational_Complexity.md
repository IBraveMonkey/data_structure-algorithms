#### 1. –ó–∞–¥–∞—á–∞ –∫–æ–º–º–∏–≤–æ—è–∂–µ—Ä–∞ (Traveling Salesman Problem, TSP)

**–î–∞–Ω–æ**: –ì–æ—Ä–æ–¥–∞ –∏ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –º–µ–∂–¥—É –Ω–∏–º–∏

```go
func tspBruteForce(cities [][]int) int {
    // cities[i][j] = —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –æ—Ç –≥–æ—Ä–æ–¥–∞ i –¥–æ –≥–æ—Ä–æ–¥–∞ j
    n := len(cities)
    minDistance := math.MaxInt
    
    // –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –≤—Å–µ –ø–µ—Ä–µ—Å—Ç–∞–Ω–æ–≤–∫–∏ –≥–æ—Ä–æ–¥–æ–≤

        distance := 0
        for i := 0; i < n-1; i++ {
            distance += cities[perm[i]][perm[i+1]]
        }
        distance += cities[perm[n-1]][perm[0]] // –í–æ–∑–≤—Ä–∞—Ç –≤ –Ω–∞—á–∞–ª–æ

            minDistance = distance
        }
    }
    
    return minDistance
}
```
